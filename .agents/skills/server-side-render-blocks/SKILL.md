# Server-Side Render Blocks

## When to Use This Skill

Creating Gutenberg blocks with:
- Server-side rendering (PHP instead of JavaScript)
- External API integration
- Mock data for demo/preview mode
- WordPress security best practices

---

## How It Works

1. **Editor (edit.js)**: User configures block settings in Gutenberg
2. **Save (save.js)**: Stores only attributes as HTML comment
3. **Render (render.php)**: Server reads attributes and outputs HTML
4. **Mock data**: Used when no API credentials are provided

---

## block.json Configuration

The key is `"render": "file:./render.php"`:

```json
{
    "$schema": "https://schemas.wp.org/trunk/block.json",
    "apiVersion": 3,
    "name": "namespace/block-name",
    "version": "0.1.0",
    "title": "Block Title",
    "category": "design",
    "textdomain": "block-name",
    "supports": {
        "html": false
    },
    "attributes": {
        "uniqueId": {
            "type": "string",
            "default": ""
        },
        "apiKey": {
            "type": "string",
            "default": ""
        },
        "placeId": {
            "type": "string",
            "default": ""
        },
        "cantidad": {
            "type": "number",
            "default": 4
        }
    },
    "editorScript": "file:./index.js",
    "editorStyle": "file:./editor.css",
    "style": "file:./style.css",
    "render": "file:./render.php"
}
```

## edit.js - Editor Component

```javascript
import { __ } from '@wordpress/i18n';
import { useBlockProps, InspectorControls } from '@wordpress/block-editor';
import { PanelBody, TextControl, RangeControl } from '@wordpress/components';
import { useEffect } from '@wordpress/element';

function generateUniqueId() {
    return 'block-' + Math.random().toString(36).substring(2, 9);
}

export default function Edit({ attributes, setAttributes, clientId }) {
    const { apiKey, placeId, cantidad, uniqueId } = attributes;

    useEffect(() => {
        if (!uniqueId) {
            setAttributes({ uniqueId: generateUniqueId() });
        }
    }, [clientId]);

    return (
        <>
            <InspectorControls>
                <PanelBody title={__('Configuration', 'textdomain')}>
                    <TextControl
                        label={__('API Key', 'textdomain')}
                        value={apiKey || ''}
                        onChange={(value) => setAttributes({ apiKey: value })}
                    />
                    <TextControl
                        label={__('Place ID', 'textdomain')}
                        value={placeId || ''}
                        onChange={(value) => setAttributes({ placeId: value })}
                    />
                    <RangeControl
                        label={__('Number of items', 'textdomain')}
                        value={cantidad || 4}
                        onChange={(value) => setAttributes({ cantidad: value })}
                        min={1}
                        max={10}
                    />
                </PanelBody>
            </InspectorControls>

            <div {...useBlockProps()}>
                {!placeId || !apiKey ? (
                    <p style={{ padding: '20px', textAlign: 'center' }}>
                        {__('Configure API credentials to display content', 'textdomain')}
                    </p>
                ) : (
                    <p style={{ padding: '20px', textAlign: 'center' }}>
                        {__('Content will be displayed on the frontend', 'textdomain')}
                    </p>
                )}
            </div>
        </>
    );
}
```

## save.js - Empty Save

```javascript
import { useBlockProps } from '@wordpress/block-editor';

export default function save() {
    return <div {...useBlockProps.save()} />;
}
```

The save function returns only the block wrapper. All frontend markup is generated by PHP.

## render.php - Server-Side Render

```php
<?php
$api_key   = isset($attributes['apiKey']) ? sanitize_text_field($attributes['apiKey']) : '';
$place_id  = isset($attributes['placeId']) ? sanitize_text_field($attributes['placeId']) : '';
$cantidad  = isset($attributes['cantidad']) ? intval($attributes['cantidad']) : 4;
$unique_id = isset($attributes['uniqueId']) ? sanitize_html_class($attributes['uniqueId']) : 'block';

$is_demo = empty($place_id) && empty($api_key);

if ($is_demo) {
    $mock_data = include __DIR__ . '/mock-data.php';
    $items = array_slice($mock_data, 0, $cantidad);
} else {
    $api_url = add_query_arg([
        'place_id' => $place_id,
        'fields'   => 'reviews',
        'key'      => $api_key,
    ], 'https://maps.googleapis.com/maps/api/place/details/json');

    $response = wp_remote_get($api_url, ['timeout' => 15]);

    if (is_wp_error($response)) {
        echo '<div class="block-error">Error: ' . esc_html($response->get_error_message()) . '</div>';
        return;
    }

    $data = json_decode(wp_remote_retrieve_body($response), true);

    if (!isset($data['result']['reviews']) || empty($data['result']['reviews'])) {
        echo '<div class="block-error">No se encontraron datos.</div>';
        return;
    }

    $items = $data['result']['reviews'];
    $items = array_slice($items, 0, $cantidad);
}
?>
<section class="block-<?php echo esc_attr($unique_id); ?>">
    <?php foreach ($items as $item) : ?>
        <div class="block-item">
            <strong><?php echo esc_html($item['author_name']); ?></strong>
            <p><?php echo esc_html($item['text']); ?></p>
        </div>
    <?php endforeach; ?>
</section>
```

## block-name.php - Plugin Registration

```php
<?php
/**
 * Plugin Name:       Block Name
 * Description:       Block description
 * Requires at least: 6.0
 * Requires PHP:      7.0
 * Version:           0.1.0
 * Text Domain:       block-name
 */

if (!defined('ABSPATH')) {
    exit;
}

function block_name_register_block() {
    $asset_file = include(plugin_dir_path(__FILE__) . 'index.asset.php');

    wp_register_script(
        'block-name-editor',
        plugins_url('index.js', __FILE__),
        $asset_file['dependencies'],
        $asset_file['version']
    );

    wp_register_style(
        'block-name-editor-style',
        plugins_url('editor.css', __FILE__),
        array(),
        $asset_file['version']
    );

    wp_register_style(
        'block-name-style',
        plugins_url('style.css', __FILE__),
        array(),
        $asset_file['version']
    );

    register_block_type(__DIR__, array(
        'editor_script' => 'block-name-editor',
        'editor_style' => 'block-name-editor-style',
        'style' => 'block-name-style',
    ));
}
add_action('init', 'block_name_register_block');
```

## Security Checklist

| Practice | Implementation |
|----------|----------------|
| Sanitization | `sanitize_text_field()`, `sanitize_email()`, `absint()` |
| Escaping | `esc_html()`, `esc_attr()`, `esc_url()` |
| Output validation | Check `isset()` before using any attribute |

```php
$cached = get_transient($cache_key);

if (false === $cached) {
    $cached = $api_data;
    set_transient($cache_key, $cached, HOUR_IN_SECONDS);
}
```

```php
if (is_wp_error($response)) {
```

## Common Issues

| Issue | Solution |
|-------|----------|
| Block not rendering | Check `"render": "file:./render.php"` in block.json |
| Mock data not showing | Verify `$is_demo` condition logic |
| API errors | Check API key and Place ID are correct |
| Styles not applying | Ensure style is registered in PHP file |
| Attributes not passed | Verify save.js returns useBlockProps.save() |

## Dynamic CSS for Server-Side Rendered Blocks

### The Problem

When using `"render": "file:./render.php"` in block.json, WordPress **completely ignores the save.js output** in the frontend. Only `render.php` is used.

This means:
- Any CSS generated in `save.js` (like `generateBlockCSS()`) **never reaches the frontend**
- The frontend only sees what `render.php` outputs

### Solution: Generate CSS in PHP

For blocks with server-side render, CSS must be generated in `render.php`:

#### 1. Use Custom Properties in SCSS

In `style.scss`, use CSS custom properties with defaults:

```scss
.apg-google-reviews-card {
    background-color: var(--apg-card-bg, #ffffff);
    color: var(--apg-card-text, #1a1a1a);
    border-radius: var(--apg-card-radius, 12px);
}
```

This provides fallback values so basic styles work even without dynamic CSS.

#### 2. Generate CSS in render.php

```php
if ( ! function_exists( 'apg_generate_block_css' ) ) {
function apg_generate_block_css( $unique_id, $attributes ) {
    $defaults = array(
        'cardBackgroundColor' => '#ffffff',
        'cardTextColor'       => '#1a1a1a',
        'cardBorderRadius'    => 12,
        // ... more attributes
    );

    $atts = array_merge( $defaults, array_intersect_key( $attributes, $defaults ) );

    $css = ".apg-google-reviews-{$unique_id} {\n";
    $css .= "    --apg-card-bg: {$atts['cardBackgroundColor']};\n";
    $css .= "    --apg-card-text: {$atts['cardTextColor']};\n";
    // ... more CSS variables
    $css .= "}\n";

    return $css;
}
}
```

#### 3. Inject CSS in render.php

```php
$block_css = apg_generate_block_css( $unique_id, $attributes );
$custom_css = isset( $attributes['customCSS'] ) ? $attributes['customCSS'] : '';
$full_css = $block_css . $custom_css;
?>
<?php if ( $full_css ) : ?>
<style><?php echo $full_css; ?></style>
<?php endif; ?>
```

#### 4. Keep save.js Simple

```javascript
import { useBlockProps, InnerBlocks } from '@wordpress/block-editor';

export default function save() {
    return (
        <div {...useBlockProps.save()}>
            <InnerBlocks.Content />
        </div>
    );
}
```

### Key Learnings

| Concept | Detail |
|---------|--------|
| save.js with render PHP | Only stores block wrapper, PHP does all frontend output |
| CSS source of truth | For server-rendered blocks, PHP is the single source |
| Custom properties | Use `var(--name, default)` in SCSS for fallback values |
| Function protection | Always use `if ( ! function_exists() )` to prevent redeclaration errors |
| Editor preview | CSS in edit.js is for preview only; won't reach frontend |
